simple description
events(A){ 
	id uid(unique id) date name shortname // 경기일정 팀이름 유니크id 등
	competitions(A) { // 참여하는팀 부가 설명
		competitors(A) {
			id // 참여한 경기의 id(위의 id랑같음)
			uid // 해당 팀 unique id
			winner // 이겼는지 졌는지 bool
			score // 팀 스코어
			team // 해당 팀 설명 내용많음
			statistics(A) // 해당 경기 stat 기록
		}
	}
	status {
		type {
			completed // 경기가 끝낫는지 안끝낫는지
		}
	}
}
(A) = array
